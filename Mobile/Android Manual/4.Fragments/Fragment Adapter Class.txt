- Used to configure the addition of fragments to a View_Pager.

- Code :
	class My_Pager_Adapter extends FragmentPagerAdapter{

	    private final List<Fragment> fragmentList = new ArrayList<>(); // list of your fragments.
	    private final List<String> titleList = new ArrayList<>(); // needed for the tabs titles(optional).
            
            // Constructor.
	    My_Pager_Adapter(FragmentManager fm) {
	        super(fm);
	    }
            
            // To specify the fragments and assign them.
	    @Override
	    public Fragment getItem(int position) {
	        return fragmentList.get(position);
	    }
            
            // To know how many fragments are there.
	    @Override
	    public int getCount() {
	        return fragmentList.size();
	    }
            
            // To add a fragment.
	    public void addFragment(Fragment fragment,String title){
	        fragmentList.add(fragment);
	        titleList.add(title); // for Tabs(optional)
	    }
            
            // To get the fragment title for the Tabs(optional).
	    @Override
	    public CharSequence getPageTitle(int position) {
	        return titleList.get(position);
	    }
	}
