- You need a view pager to show your fragments and an adapter to set them.

- You can add an appbar and a toolbar to set some tabs for your fragments.

- The Tabs can be fixed, centered, scrolled by specifying the mode in the xml file.

- Code :

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
	
	// Container of the fragments.(slide left/right to iterate).
        ViewPager viewPager = (ViewPager) findViewById(R.id.viewpager);

	// Our own adapter to add fragmnets.
        My_Pager_Adapter adapter = new My_Pager_Adapter(getSupportFragmentManager());
        adapter.addFragment(new Fragment_1(), "Tab 1"); // add an object of Fragment_1 class
        adapter.addFragment(new Fragment_2(), "Tab 2"); // add an object of a second Fragmenet class.
        adapter.addFragment(new Fragment_3(), "Tab 3"); // from a third class.
        viewPager.setOffscreenPageLimit(4); // set the limit of fragments to show.
        viewPager.setAdapter(adapter); // set the adapter to the view pager.

        // To use Tabs u need an appbar and a tab_layout inside it

        // u can delete this to just swipe for the other fragments
        // Note that u need to set the Layout_behavior of the view_pager to 'appbar_scrolling_view_behavior' to make the view_pager
        // be under the appbar since the top part of the fragments will not show if not set.
        TabLayout tabLayout = (TabLayout) findViewById(R.id.tabLayout);
        tabLayout.setupWithViewPager(viewPager); // link with the view pager.
    }
}
