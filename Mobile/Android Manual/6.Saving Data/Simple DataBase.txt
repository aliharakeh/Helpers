- Databse is used to save large amount of data and helps access parts it more smoothly.

- You are gonna use a helper class to create you database and your simple database functions.

- Helper calss:

class My_Database extends SQLiteOpenHelper {

    private final static String database_name = "my_database";
    private final static String table_name = "my_table";
    private final static String column_1 = "name"; // column_0 is the id (see below).

    My_Database(Context context) {
        super(context, database_name, null, 1);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        // the auto increment is optional you can remove it and consider the id as column_0 and change some parameters of functions
        db.execSQL("CREATE TABLE " + table_name + " (id INTEGER PRIMARY KEY AUTOINCREMENT," + column_1 +" TEXT)");
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        db.execSQL("DROP TABLE IF EXISTS " + table_name);
    }

    public boolean insert_Data(String name){
        SQLiteDatabase db = getWritableDatabase();
        ContentValues contentValues = new ContentValues();
        contentValues.put(column_1, name);
        long result = db.insert(table_name, null, contentValues);
        db.close();
        if(result == -1)
            return false;
        return true;
    }

    Cursor get_All_Data(){
        SQLiteDatabase db = getWritableDatabase();
        Cursor all_data = db.rawQuery("SELECT * from " + table_name, null);
        return all_data;
    }

    public boolean update_data(String id, String name){
        SQLiteDatabase db = getWritableDatabase();
        ContentValues contentValues = new ContentValues();
        contentValues.put(column_1, name);
        int result = db.update(table_name, contentValues, "id =?", new String[]{id});
        if(result > 0)
            return true;
        return false;
    }

    public boolean delete_data(String id){
        SQLiteDatabase db = getWritableDatabase();
        int result = db.delete(table_name, "id=?", new String[]{id});
        if(result > 0)
            return true;
        return false;
    }
}
